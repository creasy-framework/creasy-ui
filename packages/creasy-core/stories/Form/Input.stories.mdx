## Inputs

```tsx live expanded
render(() => {
  const [value1, setValue1] = React.useState<string>('');
  const [value2, setValue2] = React.useState<string>('');
  return (
    <>
      <Input label="Outlined" value={value1} onChange={(e) => setValue1(e.target.value)}/>
      <Input type="underlined" label="Underlined" value={value2} onChange={(e) => setValue2(e.target.value)}/>
    </>
  );
})
```
