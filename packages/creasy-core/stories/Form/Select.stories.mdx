## Select

```tsx live expanded
render(() => {
  const [value, setValue] = React.useState();
  const colours = [
    { value: 'r', label: 'Red' },
    { value: 'g', label: 'Green' },
    { value: 'b', label: 'Blue' },
  ];
  const onChange = (selection) => setValue(selection)

  return (
    <div style={{ height: 200 }}>
      <Select
        size="m"
        clearable={false}
        options={colours}
        label="Color"
        selectedValue={value}
        onChange={onChange} />
    </div>
  );
})
```

## Multi Select

```tsx live
render(() => {
  const [value, setValue] = React.useState();
  const colours = [
    { value: 'r', label: 'Red' },
    { value: 'g', label: 'Green' },
    { value: 'b', label: 'Blue' },
  ];
  const onChange = (selection) => {
      setValue(selection);
  }
  return (
    <div style={{ height: 200 }}>
      <Select
        size="m"
        clearable={false}
        isMultipleSelect
        options={colours}
        label="Color"
        selectedValue={value}
        onChange={onChange} />
    </div>
  );
})
```
