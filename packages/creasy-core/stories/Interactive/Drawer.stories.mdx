## Drawer

```tsx live expanded
render(() => {
  const [isShow, setIsShow] = React.useState(false);
  const close = () => setIsShow(false);
  const open = () => setIsShow(true);
  return (
    <>
      <Button color="primary" type="outlined" onClick={open}>
        Open drawer
      </Button>
      <Drawer header={
        <h3>Creasy UI</h3>
      } isShow={isShow} onClose={close}>
        <List>
          <List.Item icon={<Icons.Form />}>
            Forms
          </List.Item>
          <List.Item icon={<Icons.Chart />}>
            Reports
          </List.Item>
        </List>
      </Drawer>
    </>
  );
});
```
